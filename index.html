document.querySelector('.scroll-indicator')?.addEventListener('click', () => { document.querySelector('.project')?.scrollIntoView({ behavior: 'smooth' }); }); const overlays = Array.from(document.querySelectorAll('.project-overlay')).map(el => ({ el, section: el.closest('.project'), })); const MIN_OPACITY = 0.08; function updateOverlays() { const vh = window.innerHeight; for (const { el, section } of overlays) { const top = section.getBoundingClientRect().top; const progress = Math.min(Math.max((vh - top) / vh, 0), 1); el.style.opacity = 1 - progress * (1 - MIN_OPACITY); } } let ticking = false; function onScroll() { if (ticking) return; ticking = true; requestAnimationFrame(() => { updateOverlays(); ticking = false; }); } if (overlays.length) { updateOverlays(); window.addEventListener('scroll', onScroll, { passive: true }); window.addEventListener('resize', onScroll); }
